Vitessce now supports plugin view types and file types.

:::tip

Note that the current plugin interface is open-ended and therefore we cannot guarantee that future versions of Vitessce will not break existing plugins.
For instance, we do not make any guarantees that exported helper functions such as React data-loading hooks (e.g., `useRasterData`, `useExpressionMatrixData`, etc.) will not be removed or will have a consistent function signature across Vitessce versions.
Therefore, plugin developers should be sure to document which version(s) of Vitessce that plugins have been developed under.

If you believe a plugin view type or file type would be applicable to the wider Vitessce user base, we encourage contributing it back to the Vitessce core via a pull request, which can also facilitate us to support the view type or file type in future Vitessce versions.

:::

## Plugin view types

```js
import React, { useEffect } from 'react';
import {
  Vitessce,
  CoordinationType,
  registerPluginViewType,
  registerPluginFileType,
  TitleInfo,
  useReady,
  useUrls,
  useCoordination,
  useLoaders,
  useDescription,
} from 'vitessce';
import myViewConfig from './my-view-config';
import 'vitessce/dist/es/production/static/css/index.css';
import './index.css';

function MyPluginView(props) {
  const {
    spatialZoom,
  } = props;
  return (
    <div>
      <p>{spatialZoom}</p>
    </div>
  );
}

function MyPluginViewSubscriber(props) {
  const {
    coordinationScopes,
    removeGridComponent,
    theme,
    title = 'My plugin view',
  } = props;

  const loaders = useLoaders();

  // Get "props" from the coordination space.
  const [{
    dataset,
    spatialZoom
  }, {
    setSpatialZoom,
  }] = useCoordination(
    [
      CoordinationType.DATASET,
      CoordinationType.SPATIAL_ZOOM,
    ],
    coordinationScopes,
  );

  const [urls, addUrl, resetUrls] = useUrls();
  const [
    isReady,
    setItemIsReady,
    setItemIsNotReady, // eslint-disable-line no-unused-vars
    resetReadyItems,
  ] = useReady([]);

  // Reset file URLs and loader progress when the dataset has changed.
  useEffect(() => {
    resetUrls();
    resetReadyItems();
    // eslint-disable-next-line react-hooks/exhaustive-deps
  }, [loaders, dataset]);

  return (
    <TitleInfo
      title={title}
      theme={theme}
      removeGridComponent={removeGridComponent}
      isReady={isReady}
      urls={urls}
    >
      <MyPluginView
        spatialZoom={spatialZoom}
      />
    </TitleInfo>
  );
}

registerPluginViewType(
  'myCustomZoomController',
  MyPluginViewSubscriber,
);

export default function MyApp() {
  return (
    <Vitessce
      config={myViewConfig}
      height={800}
      theme="light"
    />
  );
}
```
